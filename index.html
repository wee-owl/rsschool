Code example

Task from Codewars «Highest and Lowest»
This assignment is given a string of space-separated numbers and must return the largest and smallest number

function highAndLow(numbers) {
   var str = numbers.replace(/ /gi, ',');
   var i = JSON.parse("[" + str + "]");
   var x = String(i.sort((a,b)=>b-a)[0]);
   var y = String(i.sort((a,b)=>b-a)[i.length-1]);
return x + ' ' + y;
}
